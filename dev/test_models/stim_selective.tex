Under the \texttt{stop} component there are three declared outcomes:

\begin{itemize}
\tightlist
\item
  \texttt{A\ =\ first\_of(inhibit(A,\ S1),\ all\_of(A,\ S1,\ inhibit(IS,\ S2)))}
\item
  \texttt{B\ =\ first\_of(inhibit(B,\ S1),\ all\_of(B,\ S1,\ inhibit(IS,\ S2)))}
\item
  \texttt{STOP\ =\ all\_of(S1,\ inhibit(S2,\ IS))}, mapped to
  \texttt{NA}
\end{itemize}

Consider the two branches of A and B:

\begin{itemize}
\tightlist
\item
  early \texttt{A}: \texttt{E\_A\ =\ inhibit(A,\ S1)}
\item
  late \texttt{A}: \texttt{L\_A\ =\ all\_of(A,\ S1,\ inhibit(IS,\ S2))}
\item
  early \texttt{B}: \texttt{E\_B\ =\ inhibit(B,\ S1)}
\item
  late \texttt{B}: \texttt{L\_B\ =\ all\_of(B,\ S1,\ inhibit(IS,\ S2))}
\end{itemize}

Then:

\begin{itemize}
\tightlist
\item
  \texttt{A\_out\ =\ first\_of(E\_A,\ L\_A)}
\item
  \texttt{B\_out\ =\ first\_of(E\_B,\ L\_B)}
\end{itemize}

For the current engine, observed \texttt{NA} is leftover visible mass
after \texttt{A} and \texttt{B}, so the hand-written likelihood targets:

\begin{itemize}
\tightlist
\item
  \texttt{p(A\_obs\ at\ t)\ =\ f\_A\_out(t)\ S\_B\_out(t)}
\item
  \texttt{p(B\_obs\ at\ t)\ =\ f\_B\_out(t)\ S\_A\_out(t)}
\item
  \texttt{P(NA\_obs)\ =\ 1\ -\ P(A\_obs)\ -\ P(B\_obs)}
\end{itemize}

\subsection{The actual formulas}\label{the-actual-formulas}

Write \texttt{f\_X(t)}, \texttt{F\_X(t)}, and \texttt{S\_X(t)} for the
density, cdf, and survival of accumulator \texttt{X}.

For this model:

\begin{itemize}
\tightlist
\item
  \texttt{E\_A\ =\ inhibit(A,\ S1)}
\item
  \texttt{E\_B\ =\ inhibit(B,\ S1)}
\item
  \texttt{H\ =\ inhibit(IS,\ S2)}
\item
  \texttt{L\_A\ =\ all\_of(A,\ S1,\ H)}
\item
  \texttt{L\_B\ =\ all\_of(B,\ S1,\ H)}
\end{itemize}

The pieces are:

\begin{itemize}
\item
  \texttt{f\_EA(t)\ =\ f\_A(t)\ S\_S1(t)}
\item
  \texttt{f\_EB(t)\ =\ f\_B(t)\ S\_S1(t)}
\item
  \texttt{f\_H(t)\ =\ f\_IS(t)\ S\_S2(t)}
\item
  \texttt{F\_EA(t)\ =\ ∫\_0\^{}t\ f\_A(u)\ S\_S1(u)\ du}
\item
  \texttt{F\_EB(t)\ =\ ∫\_0\^{}t\ f\_B(u)\ S\_S1(u)\ du}
\item
  \texttt{F\_H(t)\ =\ ∫\_0\^{}t\ f\_IS(u)\ S\_S2(u)\ du}
\item
  \texttt{S\_EA(t)\ =\ 1\ -\ F\_EA(t)}
\item
  \texttt{S\_EB(t)\ =\ 1\ -\ F\_EB(t)}
\item
  \texttt{S\_H(t)\ =\ 1\ -\ F\_H(t)}
\end{itemize}

For the late branches:

\begin{itemize}
\item
  \texttt{F\_LA(t)\ =\ F\_A(t)\ F\_S1(t)\ F\_H(t)}
\item
  \texttt{F\_LB(t)\ =\ F\_B(t)\ F\_S1(t)\ F\_H(t)}
\item
  \texttt{f\_LA(t)\ =\ f\_A(t)\ F\_S1(t)\ F\_H(t)\ +\ F\_A(t)\ f\_S1(t)\ F\_H(t)\ +\ F\_A(t)\ F\_S1(t)\ f\_H(t)}
\item
  \texttt{f\_LB(t)\ =\ f\_B(t)\ F\_S1(t)\ F\_H(t)\ +\ F\_B(t)\ f\_S1(t)\ F\_H(t)\ +\ F\_B(t)\ F\_S1(t)\ f\_H(t)}
\item
  \texttt{S\_LA(t)\ =\ 1\ -\ F\_LA(t)}
\item
  \texttt{S\_LB(t)\ =\ 1\ -\ F\_LB(t)}
\end{itemize}

The visible outcomes are:

\begin{itemize}
\tightlist
\item
  \texttt{A\_out\ =\ first\_of(E\_A,\ L\_A)}
\item
  \texttt{B\_out\ =\ first\_of(E\_B,\ L\_B)}
\end{itemize}

So:

\begin{itemize}
\item
  \texttt{f\_A\_out(t)\ =\ f\_EA(t)\ S\_LA(t)\ +\ f\_LA(t)\ S\_EA(t)}
\item
  \texttt{f\_B\_out(t)\ =\ f\_EB(t)\ S\_LB(t)\ +\ f\_LB(t)\ S\_EB(t)}
\item
  \texttt{S\_A\_out(t)\ =\ S\_EA(t)\ S\_LA(t)}
\item
  \texttt{S\_B\_out(t)\ =\ S\_EB(t)\ S\_LB(t)}
\end{itemize}

That gives the observed response densities directly:

\begin{itemize}
\tightlist
\item
  \texttt{p(A\_obs\ at\ t)\ =\ f\_A\_out(t)\ S\_B\_out(t)}
\item
  \texttt{p(B\_obs\ at\ t)\ =\ f\_B\_out(t)\ S\_A\_out(t)}
\end{itemize}

Expanding \texttt{A} into its two branches:

\begin{itemize}
\item
  early \texttt{A} term:

  \texttt{f\_EA(t)\ S\_LA(t)\ S\_EB(t)\ S\_LB(t)}
\item
  late \texttt{A} term:

  \texttt{f\_LA(t)\ S\_EA(t)\ S\_EB(t)\ S\_LB(t)}
\end{itemize}

and the total is their sum.

By symmetry:

\begin{itemize}
\item
  early \texttt{B} term:

  \texttt{f\_EB(t)\ S\_LB(t)\ S\_EA(t)\ S\_LA(t)}
\item
  late \texttt{B} term:

  \texttt{f\_LB(t)\ S\_EB(t)\ S\_EA(t)\ S\_LA(t)}
\end{itemize}

and again the total is their sum.
